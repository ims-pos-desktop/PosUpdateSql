CREATE OR ALTER VIEW [dbo].[NEPMONTH_FISCAL]
AS
SELECT A.MAHINA,B.MAHINANAME,A.SAL,MIN(AD) MINDATE, MAX(AD) MAXDATE FROM (
SELECT LEFT(MITI,2) GATE,SUBSTRING(MITI,4,2) MAHINA,SUBSTRING(MITI,7,4) SAL,AD FROM DATEMITI 
WHERE AD BETWEEN (SELECT FBDATE FROM COMPANY ) AND (SELECT FEDATE FROM COMPANY)
) AS A inner join NEPMONTH B ON A.MAHINA =B.MAHINA GROUP BY A.MAHINA,B.MAHINANAME,SAL

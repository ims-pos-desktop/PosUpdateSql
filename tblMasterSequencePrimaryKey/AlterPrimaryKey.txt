IF EXISTS (SELECT * FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE CONSTRAINT_NAME = 'PK_TBLMASTERSEQUENCES')
	AND NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.CONSTRAINT_COLUMN_USAGE WHERE CONSTRAINT_NAME = 'PK_TBLMASTERSEQUENCES' AND COLUMN_NAME = 'DIVISION')
BEGIN
ALTER TABLE tblMasterSequences DROP CONSTRAINT PK_TBLMASTERSEQUENCES
END

IF NOT  EXISTS (SELECT * FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS WHERE CONSTRAINT_NAME = 'PK_TBLMASTERSEQUENCES')
ALTER TABLE tblMasterSequences ADD CONSTRAINT PK_TBLMASTERSEQUENCES PRIMARY KEY (VNAME, DIVISION)

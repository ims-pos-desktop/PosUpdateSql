CREATE OR ALTER VIEW TRNPROD_BRANCH_OUT  
AS  
SELECT VCHRNO,CHALANNO,DIVISION, MCODE,UNIT,Quantity,RealQty, AltQty,RATE, CRATE, AMOUNT, VAT, REALRATE, ALTRATE, REALQTY_IN,ALTQTY_IN ,WAREHOUSE,PRATE,SNO,BC, ALTUNIT,PhiscalID ,STAMP, Batch  FROM RMD_TRNPRODS WHERE LEFT(VCHRNO, 2) = 'TO' 
AND (SELECT CentralizedDbMode FROM SETTING) = 0 
UNION   
SELECT VCHRNO,CHALANNO,DIVISION, MCODE,UNIT,Quantity,RealQty, AltQty,RATE, CRATE, AMOUNT, VAT, REALRATE, ALTRATE, REALQTY_IN,ALTQTY_IN ,WAREHOUSE,PRATE,SNO,BC, ALTUNIT,PhiscalID ,STAMP, Batch  FROM INVPROD WHERE LEFT(VCHRNO, 2) = 'TO' AND ADDTIONALROW = 0

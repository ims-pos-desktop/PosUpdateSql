CREATE OR ALTER PROCEDURE [dbo].[RSP_PURCHASEBOOKREPORT]
--DECLARE
	@DATE1 DATETIME,
	@DATE2 DATETIME,
	@DIVISION varchar(10)='%',
	@FX_VOUCHER_TYPE CHAR(2) = 'SI',									--VoucherType:SI
	@CHK_INCLUDE_RETURN TINYINT = 0,									--IncludeReturn:1:0
	@OPT_REPORT_TYPE TINYINT = 0,										--VoucherWise:0,DayWise:1,MonthWise:2
	@OPT_SHOWDETAL_REPORT TINYINT = 0,									--Summary:0,Detail:1
	@VCHR VARCHAR(25) = 'A',
	@SUPPLIER_ACID VARCHAR(25) = '%',
	@OPT_INAD TINYINT = 0												--AD:0,BS:1
							
AS

EXEC [dbo].[NSP_PURCHASEBOOKREPORT]
	@DATE1 =@DATE1 ,
	@DATE2 =@DATE2,
	@DIV=@DIVISION,
	@VOUCHER_TYPE =@FX_VOUCHER_TYPE,
	@INCLUDE_RETURN = @CHK_INCLUDE_RETURN,									--	0 Exclude Return   1 Include Return
	@REPORT_TYPE = @OPT_REPORT_TYPE,										--  0 VOUCHER WISE  1 DAY WISE  2 MONTH WISE	
	@SHOWDETAL_REPORT = @OPT_SHOWDETAL_REPORT,									--  0 SUMMARY REPORT  1 DETAIL REPORT
	@VCHR =@VCHR,
	@PARTY =@SUPPLIER_ACID,
	@INAD =@OPT_INAD												-- 0 ENGLISH DATE  1 NEPALI DATE

CREATE OR ALTER   PROC [dbo].[RSP_PURCHASEFIGURE_SUPPLIERWISE]
@DATE1 DATETIME,
@DATE2 DATETIME,
@DIVISION VARCHAR(3) = '%',
@OPT_SHOWSUMMARY_REPORT TINYINT = 0,								--Summary:1,Detail:0
@CHK_TOPNSORT TINYINT = 0,										--TopNSort:1:0
@SUPPLIER_ACID VARCHAR(20) ='%',
@AMOUNT  NUMERIC(18,2) = 0
AS
	EXEC SP_PURCHASEFIGURE_SUPPLIERWISE 
		@DATE1 = @DATE1,
		@DATE2=@DATE2,
		@ACID = @SUPPLIER_ACID, 
		@TOPNSORT = @CHK_TOPNSORT,
		@DIV = @DIVISION,	
		@SUMMARY = @OPT_SHOWSUMMARY_REPORT

IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = 'COSTING_DAYWISE')
CREATE TABLE COSTING_DAYWISE 
(
	TRNDATE DATE NOT NULL,
	MCODE VARCHAR(25) NOT NULL,
	RATE DECIMAL(18,8) NOT NULL,
	CONSTRAINT PK_COSTING_DAYWISE PRIMARY KEY (TRNDATE, MCODE),
	CONSTRAINT FK_COSTING_DAYWISE_MENUITEM FOREIGN KEY (MCODE) REFERENCES MENUITEM(MCODE)
)

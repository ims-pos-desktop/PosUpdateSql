IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = 'tblUserWorkingLog' AND COLUMN_NAME = 'HOSTNAME')
BEGIN
	ALTER TABLE tblUserWorkingLog ADD HostName VARCHAR(100) NULL
	ALTER TABLE tblUserWorkingLog ADD CONSTRAINT DF_TBLUSERWORKINGLOG_HOSTNAME DEFAULT (HOST_NAME()) FOR HostName
	UPDATE L SET L.HostName = S.ComputerID from tblUserWorkingLog L JOIN SESSION S ON L.UserSessId = S.SessionID
END

CREATE OR ALTER VIEW vw_OnlinePaymentModes
AS
SELECT TRNMODE
FROM	(
		SELECT 'ONLINE' TRNMODE UNION ALL
		SELECT 'ESEWA' UNION ALL
		SELECT 'FONEPAY' UNION ALL
		SELECT 'QR' UNION ALL
		SELECT 'QR MODE' UNION ALL
		SELECT 'MOCO' UNION ALL
		SELECT 'Nepal Pay' UNION ALL
		SELECT 'IME Pay' UNION ALL
		SELECT 'Smart QR' UNION ALL
		SELECT 'Khalti' UNION ALL
		SELECT 'MocoPay' UNION ALL
		SELECT 'Hamro Pay'
) AS T

CREATE OR ALTER PROCEDURE [dbo].[NSP_PLACCOUNT_REPORT]
	@DATE1 DATETIME,
	@DATE2 DATETIME,
	@DIV VARCHAR(3) = '%',
	@OSTOCK NUMERIC(22,10),
	@CSTOCK NUMERIC(22,10),
	@SUMMARY TINYINT = 1,
	@SHOWSUBLEDGER TINYINT = 0,
	@LEV INTEGER = 1000,
	@ISIVREPORT TINYINT = 1
AS

IF @ISIVREPORT = 1 
	EXEC NSP_PLACCOUNT @DATE1,@DATE2,@DIV,@OSTOCK,@CSTOCK,@SUMMARY,@SHOWSUBLEDGER,@LEV
ELSE
	EXEC NSP_PLACCOUNT_HORIZONTAL @DATE1,@DATE2,@DIV,@OSTOCK,@CSTOCK,@SUMMARY,@SHOWSUBLEDGER,@LEV

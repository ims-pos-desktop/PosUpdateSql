CREATE OR ALTER   procedure [dbo].[RSP_MontlyYearlyWeeklyCategoryise_Report]  
@flag varchar(20),
@DIV VARCHAR(25)='%'
AS
set nocount on;
--DECLARE @flag varchar(20)='Yearly',@DIV VARCHAR(25)='%'
declare @DATE1 date,@Phiscalid varchar(25)
select @Phiscalid=PhiscalID from COMPANY
SELECT @DATE1 = GETDATE()

DECLARE @DAYEND INT
SELECT @DAYEND = ENABLECOMPULSORYDAYEND  FROMÂ SETTING

if @flag='Monthly'
begin
    SELECT DATEPART(YYYY,TRNDATE) year, DATEPART(MM, TRNDATE) month, DATEPART(DD, TRNDATE) day, ISNULL(cash, 0) cash, isnull(bank, 0) bank, isnull(cash, 0) + isnull(bank, 0) totalSales FROM
    (
		SELECT IIF(@DAYEND = 0 , TRNDATE , TRN_DATE) TRNDATE, CASE WHEN TRNMODE = 'CASH' THEN 'CASH' ELSE 'BANK' END PaymentMode, CONVERT(DECIMAL(18,2), SUM(CASE WHEN VoucherType  = 'CN' THEN NETAMNT * -1 ELSE NETAMNT END)) TOTALSALES
		FROM RMD_TRNMAIN
		WHERE VoucherType IN ('SI', 'TI', 'CN') AND IIF(@DAYEND = 0 , TRNDATE , TRN_DATE) between dateadd(d,-29,@DATE1) and  @DATE1 and DIVISION like @DIV
		GROUP BY IIF(@DAYEND = 0 , TRNDATE , TRN_DATE) , CASE WHEN TRNMODE = 'CASH' THEN 'CASH' ELSE 'BANK' END
    ) A PIVOT (SUM(totalSales) FOR PAYMENTMODE IN (CASH, BANK)) P
end

else if @flag='Weekly'
begin
	SELECT DATEPART(YYYY,TRNDATE) year, DATEPART(MM, TRNDATE) month, DATEPART(DD, TRNDATE) day, ISNULL(cash, 0) cash, isnull(bank, 0) bank, isnull(cash, 0) + isnull(bank, 0) totalSales FROM
    (
		SELECT IIF(@DAYEND = 0 , TRNDATE , TRN_DATE) TRNDATE , CASE WHEN TRNMODE = 'CASH' THEN 'CASH' ELSE 'BANK' END PaymentMode, CONVERT(DECIMAL(18,2), SUM(CASE WHEN VoucherType  = 'CN' THEN NETAMNT * -1 ELSE NETAMNT END)) TOTALSALES
		FROM RMD_TRNMAIN
		WHERE VoucherType IN ('SI', 'TI', 'CN') AND IIF(@DAYEND = 0 , TRNDATE , TRN_DATE) between dateadd(d,-6,@DATE1) and @DATE1 and DIVISION like @DIV
		GROUP BY IIF(@DAYEND = 0 , TRNDATE , TRN_DATE), CASE WHEN TRNMODE = 'CASH' THEN 'CASH' ELSE 'BANK' END
    ) A
    PIVOT (SUM(totalSales) FOR PAYMENTMODE IN (CASH, BANK)) P
end

else if @flag='Yearly'
begin
	SELECT DATEPART(YYYY,IIF(@DAYEND = 0 , TRNDATE , TRN_DATE)) year, DATEname(month, IIF(@DAYEND = 0 , TRNDATE , TRN_DATE)) month, CASE WHEN TRNMODE = 'CASH' THEN 'CASH' ELSE 'BANK' END PaymentMode, CONVERT(DECIMAL(18,2), SUM(CASE WHEN VoucherType  = 'CN' THEN NETAMNT * -1 ELSE NETAMNT END)) totalSales
	FROM RMD_TRNMAIN
	WHERE VoucherType IN ('SI', 'TI', 'CN') AND PhiscalID = @Phiscalid and DIVISION like @DIV
	GROUP BY DATEPART(YYYY,IIF(@DAYEND = 0 , TRNDATE , TRN_DATE)), datename(MM, IIF(@DAYEND = 0 , TRNDATE , TRN_DATE)),DATEPART(MM, IIF(@DAYEND = 0 , TRNDATE , TRN_DATE)), CASE WHEN TRNMODE = 'CASH' THEN 'CASH' ELSE 'BANK' END
	ORDER BY YEAR, DATEPART(MM, IIF(@DAYEND = 0 , TRNDATE , TRN_DATE))
end

else if @flag='Categorywise'
begin
	SELECT MI.MCAT1 category, CONVERT(DECIMAL(18,2), SUM(CASE WHEN TM.VoucherType  = 'CN' THEN NETAMNT * -1 ELSE NETAMNT END)) totalSales
	FROM RMD_TRNMAIN TM
    JOIN RMD_TRNPROD TP ON TM.VCHRNO = TP.VCHRNO
    JOIN MENUITEM MI ON TP.MCODE = MI.MCODE
	WHERE TM.VoucherType IN ('SI', 'TI', 'CN') AND IIF(@DAYEND = 0 , TRNDATE , TRN_DATE) between dateadd(d,-29,@DATE1) and @DATE1 and tm.DIVISION like @DIV
	GROUP BY MCAT1
end

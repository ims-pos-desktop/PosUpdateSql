CREATE OR ALTER PROCEDURE [dbo].[RSP_SUPPLIERVSPRODUCT]
	@DATE1 DATETIME,
	@DATE2 DATETIME,
	@SUPPLIER_ACID VARCHAR(25),
	@OPT_FLAG TINYINT = 1,							--Summary:1,Detail:2
	@DIVISION VARCHAR(3) = '%',
	@CHK_EXCLUDEVAT TINYINT = 0              --ExcludeVat:1:0
AS
IF @CHK_EXCLUDEVAT = 0 
	EXEC SP_SUPPLIERVSPRODUCT
		@DATE1= @DATE1,
		@DATE2 = @DATE2,
		@PARTY = @SUPPLIER_ACID,
		@FLG = @OPT_FLAG,
		@DIV = @DIVISION
ELSE
	EXEC SP_SUPPLIERVSPRODUCT_EXVAT
		@DATE1= @DATE1,
		@DATE2 = @DATE2,
		@PARTY = @SUPPLIER_ACID,
		@FLG = @OPT_FLAG,
		@DIV = @DIVISION

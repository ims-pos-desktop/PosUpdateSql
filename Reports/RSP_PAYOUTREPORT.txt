CREATE OR ALTER PROCEDURE [dbo].[RSP_PAYOUTREPORT]
	@DATE1 DATE,
	@DATE2 DATE,
	@DIVISION VARCHAR(3) = '%',
	@TERMINAL VARCHAR(3) = '%',
	@USER VARCHAR(50) = '%'
AS
SELECT D.NAME BRANCH, T.NAME TERMINAL, C.TRNDATE, M.MITI, C.TRNTIME, C.SHIFTID, C.TID, C.TRNUSER, C.TRNFOR, C.AMOUNT, C.REMARKS  from cashflowtran C
JOIN DIVISION D ON C.division = d.INITIAL
JOIN DATEMITI M ON C.TRNDATE = M.AD
JOIN Session S ON C.SHIFTID = S.SessionID AND C.PhiscalID = S.PhiscalID
JOIN SALESTERMINAL T ON C.TERMINAL = T.INITIAL
WHERE C.TRNDATE BETWEEN @DATE1 AND @DATE2 AND C.division LIKE @DIVISION AND C.TERMINAL LIKE @TERMINAL AND C.TRNUSER LIKE @USER

CREATE OR alter PROCEDURE [dbo].[RSP_BANKSTATEMENT]
--DECLARE
@DATE1 AS DATETIME = '01-01-2004',
@DATE2 AS DATETIME = '01-01-2004',
@ACID AS VARCHAR(25) = '%',
@DIVISION AS VARCHAR(3) = '%',
@FLG AS TINYINT=1,
@OPT_SHOWSUMMARY_REPORT AS TINYINT=1
AS

IF @OPT_SHOWSUMMARY_REPORT=1
EXEC [dbo].[SP_SUMMARYLEDGER]
@DATE1 =@DATE1 ,
@DATE2 =@DATE2 ,
@DIV=@DIVISION,
@ACID=@ACID

ELSE
EXEC [dbo].[SP_ACLEDGER_PROC]
@DATE1 =@DATE1 ,
@DATE2 =@DATE2 ,
@DIV=@DIVISION,
@ACID=@ACID,
@FLG=@FLG

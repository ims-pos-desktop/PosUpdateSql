CREATE  OR ALTER   PROCEDURE [dbo].[RSP_TRIALBALANCE]
--DECLARE 
	@DATE1 DATETIME,
	@DATE2 DATETIME,
	@DIVISION VARCHAR(3) = '%',
	@COSTCENTER VARCHAR(50)='%',
	@CHK_SHOWSUBLEDGER TINYINT = 0,                  --ShowSubLedger:1:0
	@SHOWDEBTORSCREDITORS TINYINT= 0,
	@CHK_ONLYOPENINGTRIAL TINYINT = 0,						--OnlyOpening:1:0
	@OPT_SHOWSUMMARY_REPORT TINYINT = 0,                    --Summary:1,Detail:0
	@GroupLevel TINYINT = 1,
	@OPT_LEDGERWISE TINYINT = 0,                   --GroupWise:0,LedgerWise:1
	@CHK_SHOWZEROBL TINYINT = 0,					  --ShowZeroBal:1:0
	@FYID VARCHAR(10) = '%',
	@OSTOCK NUMERIC(22,10) = 0,
	@CSTOCK NUMERIC(22,10) = 0
AS
exec [dbo].[NSP_TRIALBALANCE]
	@DATE1 = @DATE1,
	@DATE2 = @DATE2,
	@DIV = @DIVISION,
	@COSTCENTER = @COSTCENTER,
	@SHOWSUBLEDGER = @CHK_SHOWSUBLEDGER,
	@SHOWDEBTORSCREDITORS = @SHOWDEBTORSCREDITORS,
	@ONLYOPENINGTRIAL = @CHK_ONLYOPENINGTRIAL,
	@SUMMARYREPORT = @OPT_SHOWSUMMARY_REPORT,
	@LEV = @GroupLevel,
	@LEDGERWISE = @OPT_LEDGERWISE,
	@SHOWZEROBL = @CHK_SHOWZEROBL,
	@FYID = @FYID,
	@OSTOCK = @OSTOCK,
	@CSTOCK = @CSTOCK

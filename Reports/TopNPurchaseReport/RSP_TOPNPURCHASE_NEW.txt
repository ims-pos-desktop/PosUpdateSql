CREATE OR ALTER     PROCEDURE [dbo].[RSP_TOPNPURCHASE_NEW]
	@MGROUP VARCHAR(15)= '%',
	@OPT_RMODE TINYINT = 1,    --Qty:1,Value:2
	@DIVISION VARCHAR(3),
	@DATE1 DATETIME,
	@DATE2 DATETIME,
	@OPT_WISE TINYINT = 1,     --Item:1,Mgroup:2,Supplier:3
	@VCHR VARCHAR(10) = 'X',
	@SUPPLIER_ACID VARCHAR(50) = '%'	
AS
    EXEC SP_TOPNPURCHASE_NEW
	@MGROUP = @MGROUP,
	@RMODE = @OPT_RMODE,
	@DIV = @DIVISION,
	@DATE1 = @DATE1,
	@DATE2 = @DATE2,
	@FLAG = @OPT_WISE,
	@PARTY = @SUPPLIER_ACID

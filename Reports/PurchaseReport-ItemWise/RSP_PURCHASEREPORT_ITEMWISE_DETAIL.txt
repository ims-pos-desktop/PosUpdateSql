CREATE OR ALTER PROCEDURE [dbo].[RSP_PURCHASEREPORT_ITEMWISE_DETAIL]
--DECLARE 
	@DATE1 DATETIME,
	@DATE2 DATETIME,
	@DIVISION VARCHAR(3) = '%',
	@MCODE VARCHAR(25) = '%',
	@BARCODE VARCHAR(MAX)= '%',	
	@CHK_BARCODEWISEREPORT TINYINT = 0,        --CHK_BARCODEWISEREPORT:1:0
	@CHK_BARCODEDETAIL TINYINT =0,             --CHK_BARCODEDETAIL:1:0
	@CHK_DIVISIONWISEREPORT TINYINT = 0,       --CHK_DIVISIONWISEREPORT:1:0
	@OPT_WISE VARCHAR(25) = 'BILLWISE',        --BillWise:BILLWISE,DayWise:DAYWISE
	@SHOWINTERCOMPANYONLY TINYINT = 0
AS
    EXEC NSP_PURCHASEREPORT_ITEMWISE_DETAIL
	@DATE1 = @DATE1,
	@DATE2 = @DATE2,
	@DIV = @DIVISION,
	@MC = @MCODE,
	@BC = @BARCODE,
	@BARCODEWISEREPORT = @CHK_BARCODEWISEREPORT,
	@BARCODEDETAIL = @CHK_BARCODEDETAIL,
	@DIVISIONWISEREPORT = @CHK_DIVISIONWISEREPORT,   
	@WISE = @OPT_WISE

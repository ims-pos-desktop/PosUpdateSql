CREATE OR ALTER   PROCEDURE [dbo].[RSP_TRANSFER_INOUT_REPORT]
  --DECLARE
	@DATE1 DATETIME,
	@DATE2 DATETIME,
	@DIVISION VARCHAR(3) = '%',
	@FROM_BRANCH VARCHAR(25) = '%',	
	@OPT_BRANCHINOUT VARCHAR(3) = 'TR%',        --BranchTransferOut:TO,BranchTransferIn:TR
	@OPT_SHOWSUMMARY_REPORT TINYINT = 1         --Summary:1,Detail:0
AS
IF @OPT_SHOWSUMMARY_REPORT= 1
	SET @OPT_SHOWSUMMARY_REPORT = 0
ELSE 
	SET @OPT_SHOWSUMMARY_REPORT = 1
 EXEC NSP_TRANSFER_INOUT_REPORT
	@DATE1 = @DATE1,
	@DATE2 = @DATE2,
	@DIV = @DIVISION,
	@BRANCH = @FROM_BRANCH,
	@VNO  = @OPT_BRANCHINOUT,
	@DETAILREPORT = @OPT_SHOWSUMMARY_REPORT

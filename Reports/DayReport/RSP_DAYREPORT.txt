CREATE OR ALTER   PROCEDURE [dbo].[RSP_DAYREPORT]
	@DATE1 DATETIME,
	@DATE2 DATETIME,
	@DIVISION VARCHAR(3) = '%',
	@USER VARCHAR(50) = '%',
	@OPT_ENTRYDATEWISE TINYINT = 0,             --DateWise:1,VoucherWise:0
	@FROMVNO NUMERIC(18),
	@TOVNO NUMERIC(18),
	@VTYPE VARCHAR(25),
	@SERIES VARCHAR(25),
	@OPT_SHOWSUMMARY_REPORT TINYINT = 0         --Summary:1,Detail:0
AS
DECLARE @DETAIL TINYINT = 0
IF @OPT_SHOWSUMMARY_REPORT = 0
	SET @DETAIL = 1
EXEC NSP_DAYREPORT 
     @DATE1 = @DATE1,
	 @DATE2 = @DATE2,
	 @DIVISION = @DIVISION,
	 @USER= @USER,
	 @ENTRYDATEWISE = @OPT_ENTRYDATEWISE,
	 @FROMVNO = @FROMVNO,
	 @TOVNO = @TOVNO,
	 @VTYPE = @VTYPE,
	 @SERIES = @SERIES,
	 @DETAILREPORT = @DETAIL

CREATE OR ALTER   PROCEDURE [dbo].[NSP_BALANCESHEET_REPORT]
--DECLARE 
	@DATE DATETIME,
	@DIV VARCHAR(3) = '%',
	@OSTOCK NUMERIC(22,10),
	@CSTOCK NUMERIC(22,10),
	@SUMMARY TINYINT = 0,
	@LEV INTEGER = 2,
	@ISVREPORT TINYINT = 1,
	@FYID VARCHAR(10) = '%',
	@IsUserInput BIT= 0                       --IsUserInput:1:0
AS

IF @ISVREPORT = 1
	EXEC NSP_BALANCESHEET @DATE,@DIV,@OSTOCK,@CSTOCK,@SUMMARY,@LEV, @FYID,	@IsUserInput 
ELSE	
	EXEC NSP_BALANCESHEET_HORIZONTAL  @DATE,@DIV,@OSTOCK,@CSTOCK,@SUMMARY,@LEV, @FYID, 	@IsUserInput

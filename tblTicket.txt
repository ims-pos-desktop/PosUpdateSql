CREATE TABLE tblTicket
(
	ticketNo VARCHAR(50) NOT NULL PRIMARY KEY,
	VCHRNO VARCHAR(25) NOT NULL,
	DIVISION CHAR(3) NOT NULL,
	PhiscalID VARCHAR(20) NOT NULL,
	MCODE VARCHAR(25) NOT NULL,	
	ticketName VARCHAR(500) NOT NULL,
	stamp DATETIME NOT NULL,
	isUsed BIT NOT NULL,
	isCancelled BIT NOT NULL,
	cancelBillNo VARCHAR(25) NULL,
	CONSTRAINT FK_tblTicket_MenuItem FOREIGN KEY (MCODE) REFERENCES MENUITEM(MCODE),
	CONSTRAINT FK_tblTicket_PhiscalYear FOREIGN KEY (PhiscalId) REFERENCES PhiscalYear(PhiscalId),
	CONSTRAINT FK_tblTicket_DIVISION FOREIGN KEY (DIVISION) REFERENCES DIVISION(INITIAL)	
)

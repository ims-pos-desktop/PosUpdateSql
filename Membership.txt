IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = 'CustomerProfile' AND COLUMN_NAME = 'ReferredBy')
ALTER TABLE CustomerProfile ADD ReferredBy VARCHAR(50)

IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = 'Membership' AND COLUMN_NAME = 'RegMobile')
ALTER TABLE Membership ADD RegMobile VARCHAR(50)

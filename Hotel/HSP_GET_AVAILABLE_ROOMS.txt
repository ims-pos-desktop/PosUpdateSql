CREATE OR ALTER  PROC HSP_GET_AVAILABLE_ROOMS
--DECLARE
@CheckInDate DATETIME,
@CheckOutDate DATETIME,
@CategoryId INT
AS
--SET @CheckInDate = '2019-12-30'
--SET @CheckOutDate = '2019-12-31'
--SET @CategoryId = 1
SELECT A.* FROM HTL_ROOM  A LEFT JOIN
(
	SELECT CategoryId, RoomId FROM
	(
		SELECT CategoryId, RoomId, CheckInDate, CheckOutDate FROM HTL_BOOKING B JOIN HTL_BOOKING_STATUS BS ON B.BookingId = BS.BookingId AND B.DIVISION = BS.DIVISION AND B.PHISCALID = BS.PHISCALID
		WHERE BS.STATUS = 'BOOKED' AND  (@CheckInDate BETWEEN CHECKINDATE AND DATEADD(D, -1, CHECKOUTDATE) OR @CheckOutDate BETWEEN DATEADD(D, 1, CheckInDate) AND CheckOutDate)
		UNION ALL
		SELECT CategoryId, RoomId, CheckInDate, CheckOutDate FROM HTL_CHECKIN_DETAILS B JOIN HTL_CHECKIN_STATUS BS ON B.CheckInId = BS.CheckInId AND B.DIVISION = BS.DIVISION AND B.PHISCALID = BS.PHISCALID
		WHERE BS.STATUS = 'CHECKED IN' --AND  (@CheckInDate BETWEEN CHECKINDATE AND DATEADD(D, -1, CHECKOUTDATE) OR @CheckOutDate BETWEEN DATEADD(D, 1, CheckInDate) AND CheckOutDate)
	) A 
) B ON A.CategoryId = B.CategoryId AND A.RoomId = B.RoomId
WHERE (@CategoryId = 0 OR A.CategoryId = @CategoryId) AND B.CategoryId IS NULL

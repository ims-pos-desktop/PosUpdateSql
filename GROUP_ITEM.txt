IF NOT EXISTS(SELECT * FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = 'GROUP_ITEM')
CREATE TABLE [dbo].[GROUP_ITEM]
(
	[GROUPID] [varchar](50) NOT NULL,
	[MCODE] [varchar](25) NOT NULL
	CONSTRAINT PK_GROUP_ITEM PRIMARY KEY (GROUPID, MCODE),
	CONSTRAINT FK_GROUP_ITEM_MENUITEM FOREIGN KEY (MCODE) REFERENCES MENUITEM(MCODE)
)

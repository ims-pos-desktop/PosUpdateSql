IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = 'RateGroup_Master')
CREATE TABLE RateGroup_Master
(
	RateGroup_ID INT NOT NULL,
	RateGroup_Name VARCHAR(100) NOT NULL,
	RateGroup_Type VARCHAR(20) NOT NULL,
	Discontinued BIT NOT NULL,
	IsDeleted BIT NOT NULL,
	Created DATETIME NOT NULL,
	CreatedBy VARCHAR(50) NOT NULL,
	LastModified DATETIME NULL,
	LastModifiedBy VARCHAR(50) NULL,
	Deleted DATETIME NULL,
	DeletedBy VARCHAR(50) NULL,
	CONSTRAINT PK_RateGroupMaster PRIMARY KEY (RateGroup_ID),
)

IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = 'RateGroup_Detail')
CREATE TABLE RateGroup_Detail
(
	RateGroup_ID INT NOT NULL,
	MGroup VARCHAR(25) NULL,
	Parent VARCHAR(25) NULL,
	MCode VARCHAR(25) NULL,
	MCat VARCHAR(25) NULL,
	Rate Decimal(5,2) NOT NULL,
	Amount Decimal(18,12) NOT NULL,
	CONSTRAINT FK_RateGroupDetail FOREIGN KEY (RateGroup_ID) REFERENCES RateGroup_Master(RateGroup_ID)
)

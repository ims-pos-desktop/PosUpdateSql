IF EXISTS( SELECT PhiscalID, FEDATE FROM COMPANY WHERE PhiscalID = '78/79' AND CONVERT(DATE, GETDATE()) > CONVERT(DATE, FEDATE))
BEGIN
	BEGIN TRANSACTION
	INSERT INTO PHISCALYEAR(PhiscalID, BeginDate, EndDate, Stamp) VALUES('79/80', '17 JUL 2022', '16 JUL 2023', CONVERT(FLOAT, GETDATE()))
	UPDATE COMPANY SET  FBDATE = '17 JUL 2022', FEDATE = '16 JUL 2023', FBDATE_BS = '01/04/2079', FEDATE_BS = '31/04/2080', PhiscalID = '79/80'
	COMMIT TRANSACTION
END

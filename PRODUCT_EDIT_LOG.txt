IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = 'LOGMENUITEM')
 BEGIN
 SELECT * INTO LOGMENUITEM FROM MENUITEM WHERE 0 = 1
 ALTER TABLE LOGMENUITEM ADD EDITSTAMP FLOAT NOT NULL, ACTION VARCHAR(10) NOT NULL, LOGUSER VARCHAR(100) NOT NULL
 END

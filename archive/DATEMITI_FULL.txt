CREATE OR ALTER VIEW DATEMITI_FULL
AS
SELECT LEFT(MITI,2) GATE,SUBSTRING(MITI,4,2) MAHINA,B.MAHINANAME,SUBSTRING(MITI,7,4) SAL,AD,MITI,
MIN(AD) OVER (PARTITION BY SUBSTRING(MITI,7,4),SUBSTRING(MITI,4,2)) MINDATE,
MAX(AD) OVER (PARTITION BY SUBSTRING(MITI,7,4),SUBSTRING(MITI,4,2)) MAXDATE  FROM DATEMITI A
LEFT JOIN NEPMONTH B  ON SUBSTRING(MITI,4,2) = B.MAHINA

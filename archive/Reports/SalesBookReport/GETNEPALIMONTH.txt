CREATE OR ALTER FUNCTION [dbo].[GETNEPALIMONTH]  (@MON AS INTEGER)
RETURNS varchar(25) AS  
BEGIN 

DECLARE @MONTH AS VARCHAR(25)
BEGIN
IF @MON = 1
	SET @MONTH =  'BAISHAKH'
ELSE IF @MON = 2
	SET @MONTH ='JESTHA'
ELSE IF @MON = 3
	SET @MONTH =  'ASHAD'
ELSE IF @MON = 4
	SET @MONTH ='SHRAWAN'
ELSE IF @MON = 5
	SET @MONTH ='BHADRA'
ELSE IF @MON = 6
	SET @MONTH ='ASHWIN'
ELSE IF @MON = 7
	SET @MONTH = 'KARTIK'
ELSE IF @MON = 8
	RETURN 'MANSIR'
ELSE IF @MON = 9
	SET @MONTH = 'POUSH'
ELSE IF @MON = 10
	SET @MONTH = 'MAGH'
ELSE IF @MON = 11
	SET @MONTH = 'FALGUN'
ELSE IF @MON = 12
	SET @MONTH ='CHAITRA'
END
RETURN @MONTH
END

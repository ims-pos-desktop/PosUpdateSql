CREATE OR ALTER PROCEDURE [dbo].[RSP_BALANCESHEET_REPORT]
--DECLARE 
	@DATE2 DATETIME,
	@DIVISION VARCHAR(3) = '%',
	@OSTOCK NUMERIC(22,10),
	@CSTOCK NUMERIC(22,10),
	@OPT_SHOWSUMMARY_REPORT TINYINT = 0,                    --Summary:1,Detail:0
	@GroupLevel INTEGER = 2,
	@OPT_ISVERTICAL_REPORT TINYINT = 1                      --vertical:1,horizontal:0
AS
EXEC NSP_BALANCESHEET_REPORT @DATE = @DATE2, 
@DIV = @DIVISION,
@OSTOCK = @OSTOCK,
@CSTOCK = @CSTOCK,
@SUMMARY = @OPT_SHOWSUMMARY_REPORT,
@LEV = @GroupLevel,
@ISVREPORT = @OPT_ISVERTICAL_REPORT

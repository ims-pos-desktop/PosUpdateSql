CREATE OR ALTER   PROCEDURE [dbo].[RSP_JOURNALBOOK]
--DECLARE
	@DATE1 DATETIME,
	@DATE2 DATETIME,
	@DIVISION varchar(10)='%',
	@SERIES VARCHAR(15) = '%',								
	@CCENTER VARCHAR(50) = '%',									
	@VNO1 FLOAT = 0,									
	@VNO2 FLOAT = 0,
	@OPT_REPMODE TINYINT = 0									--DateWise:0,VoucherWise:1
AS

EXEC NSP_JOURNALBOOK 
	@DATE1 = @DATE1,
	@DATE2 = @DATE2,
	@DIV = @DIVISION,
	@SERIES = @SERIES,
	@CCENTER = @CCENTER,
	@VNO1 = @VNO1,
	@VNO2 = @VNO2,
	@REPMODE = @OPT_REPMODE

IF OBJECT_ID('KOTCATEGORY') IS NOT NULL
BEGIN
	IF OBJECT_ID('PK_KOTCATEGORY') IS NULL
	ALTER TABLE KOTCATEGORY ADD CONSTRAINT PK_KOTCATEGORY PRIMARY KEY (ID)

	IF EXISTS(SELECT DISTINCT ISBARITEM FROM MENUITEM A LEFT JOIN KOTCATEGORY B ON A.ISBARITEM = B.ID WHERE ISBARITEM IS NOT NULL AND B.ID IS NULL)
	INSERT INTO KOTCATEGORY
	SELECT DISTINCT ISBARITEM, 'N/A' FROM MENUITEM A LEFT JOIN KOTCATEGORY B ON A.ISBARITEM = B.ID WHERE ISBARITEM IS NOT NULL AND B.ID IS NULL

	IF OBJECT_ID('FK_MENUITEM_KOTCATEGORY') IS NULL
	ALTER TABLE MENUITEM ADD CONSTRAINT FK_MENUITEM_KOTCATEGORY FOREIGN KEY (IsBarItem) REFERENCES KOTCATEGORY(ID)
END

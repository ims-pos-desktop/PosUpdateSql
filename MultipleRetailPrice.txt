IF NOT EXISTS(SELECT * FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = 'MultipleRetailPrice')
CREATE TABLE MultipleRetailPrice 
(
	Id INT NOT NULL IDENTITY(1, 1) PRIMARY KEY,
    	MCODE varchar(25) FOREIGN KEY REFERENCES MENUITEM(MCODE),
    	RATE numeric(20,12) NOT NULL,
	EXPDATE date ,
	ISACTIVE tinyint ,
	CreatedBy varchar(100),
	STAMP datetime	
);

IF NOT EXISTS(SELECT * FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = 'MultipleRetailPrice' AND COLUMN_NAME = 'Id')
ALTER TABLE MultipleRetailPrice ADD Id INT NOT NULL IDENTITY(1, 1) PRIMARY KEY

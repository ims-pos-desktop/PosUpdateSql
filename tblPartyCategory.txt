IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = 'tblPartyCategory')
CREATE TABLE tblPartyCategory
(
	CATEGORY_ID INT NOT NULL IDENTITY,
	CATEGORY_NAME VARCHAR(100) NOT NULL,
	CONSTRAINT PK_tblPartyCategory PRIMARY KEY (CATEGORY_ID)
)

IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_NAME = 'tblArea')
CREATE TABLE tblArea
(
	AREA_ID INT NOT NULL IDENTITY,
	AREA_NAME VARCHAR(100) NOT NULL,
	CONSTRAINT PK_tblArea PRIMARY KEY (AREA_ID)
)

IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = 'RMD_ACLIST' AND COLUMN_NAME = 'AREA_ID')
ALTER TABLE RMD_ACLIST ADD AREA_ID INT

IF NOT EXISTS (SELECT * FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = 'RMD_ACLIST' AND COLUMN_NAME = 'CATEGORY_ID')

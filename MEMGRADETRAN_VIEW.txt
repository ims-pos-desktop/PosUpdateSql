CREATE OR ALTER VIEW [dbo].[MEMGRADETRAN_VIEW] WITH SCHEMABINDING
AS
SELECT VCHRNO,DIVISION,MEMBERID,AMOUNT,MEMBERPOINT MEMTOTPOINT, 'Opening' Remarks, PhiscalID, STAMP, 0 MEMREDEEMPOINT,TRNDATE, NULL COMPANY FROM dbo.MEMGRADETRANOPENING
UNION ALL
SELECT VCHRNO, DIVISION, MEMBERID, AMOUNT, MEMTOTPOINT, REMARKS, PhiscalID, STAMP, MEMREDEEMPOINT, TRNDATE, COMPANY FROM dbo.MEMGRADETRAN

IF EXISTS (SELECT * FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = 'TRNTRAN' AND COLUMN_NAME = 'A_ACID' AND IS_NULLABLE = 'YES')
	AND NOT EXISTS(SELECT * FROM sys.check_constraints WHERE [NAME] = 'CHECK_TRNTRAN_AACID')
BEGIN
	ALTER TABLE [dbo].[TRNTRAN]  WITH NOCHECK ADD  CONSTRAINT [CHECK_TRNTRAN_AACID] CHECK  (A_ACID IS NOT NULL) 
	ALTER TABLE [dbo].[TRNTRAN] CHECK CONSTRAINT [CHECK_TRNTRAN_AACID]
END
